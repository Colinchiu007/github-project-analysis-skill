private signOAuth1(method: string, url: string, accessToken: string, accessSecret: string): string {
  const params: Record<string, string> = {
    oauth_consumer_key: process.env.X_API_KEY!,
    oauth_nonce: randomBytes(16).toString('hex'),
    oauth_signature_method: 'HMAC-SHA1',
    oauth_timestamp: String(Math.floor(Date.now() / 1000)),
    oauth_token: accessToken,
    oauth_version: '1.0',
  };
  const paramString = Object.keys(params).sort()
    .map((k) => `${pct(k)}=${pct(params[k])}`).join('&');
  const baseString = [method.toUpperCase(), pct(url.split('?')[0]), pct(paramString)].join('&');
  const signingKey = `${pct(process.env.X_API_SECRET!)}&${pct(accessSecret)}`;
  params.oauth_signature = createHmac('sha1', signingKey).update(baseString).digest('base64');
  return 'OAuth ' + Object.keys(params).sort()
    .map((k) => `${pct(k)}="${pct(params[k])}"`).join(', ');
}

private async uploadPicture(fileName, accessToken, personId, picture, type = 'personal') {
  if (isVideo) {
    const sendUrlRequest = `https://api.linkedin.com/rest/videos?action=initializeUpload`;
    const { uploadUrl, video } = await (await this.fetch(sendUrlRequest, {
      method: 'POST',
      headers: { Authorization: `Bearer ${accessToken}`, 'LinkedIn-Version': '202601' },
      body: JSON.stringify({ initializeUploadRequest: { fileSizeBytes: picture.length, uploadParts: Math.ceil(picture.length / (1024 * 1024 * 2)) } }),
    })).json();
    const etags = [];
    for (let i = 0; i < picture.length; i += 1024 * 1024 * 2) {
      const upload = await this.fetch(`${uploadUrl}?partNumber=${i / (1024 * 1024 * 2) + 1}&uploadId=`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/octet-stream', Authorization: `Bearer ${accessToken}` },
        body: picture.slice(i, i + 1024 * 1024 * 2),
      });
      etags.push(upload.headers.get('etag'));
    }
    await this.waitForMediaToBeReady(video, accessToken, 'videos');
  }
}

private async waitForMediaToBeReady(urn, accessToken, type, maxAttempts = 20, intervalMs = 30000) {
  for (let attempt = 0; attempt < maxAttempts; attempt++) {
    const json = await (await this.fetch(
      `https://api.linkedin.com/rest/${type}/${encodeURIComponent(urn)}`,
      { method: 'GET', headers: { 'LinkedIn-Version': '202601', Authorization: `Bearer ${accessToken}` } }
    )).json();
    if (json.status === 'AVAILABLE') return;
    if (json.status === 'PROCESSING_FAILED') throw new BadBody(...);
    await timer(intervalMs);
  }
  throw new BadBody(..., `Timed out waiting for LinkedIn ${type} to be ready`);
}
```

def _publish_single(self, access_token, ig_user_id, content):
    payload = {}
    if content.post_type == PostType.REEL:
        payload["media_type"] = "REELS"
        payload["video_url"] = content.media_urls[0]
    elif content.post_type == PostType.STORY:
        payload["media_type"] = "STORIES"
        payload["image_url" if not content.media_urls[0].endswith(('.mp4','.mov')) else "video_url"] = content.media_urls[0]
    else:
        payload["image_url"] = content.media_urls[0]
    if content.text:
        payload["caption"] = content.text
    container_id = self._create_container(access_token, ig_user_id, payload)
    self._wait_for_container(access_token, container_id)
    return self._publish_container(access_token, ig_user_id, container_id)

def _wait_for_container(self, access_token, container_id):
    for _ in range(CONTAINER_POLL_MAX_ATTEMPTS):
        resp = self._request("GET", f"{BASE_URL}/{container_id}", access_token=access_token,
            params={"fields": "status_code,status"})
        data = resp.json()
        if data.get("status_code") == "FINISHED": return
        if data.get("status_code") == "ERROR":
            raise PublishError(f"Instagram container failed: {data.get('status', 'unknown error')}")
        time.sleep(CONTAINER_POLL_INTERVAL)
    raise PublishError("Instagram container processing timed out")
```

### 2.1 OAuth2 PKCE 流程（TikTok）

```python
def _pkce_code_challenge(code_verifier: str) -> str:
    """TikTok 偏离 RFC 7636：challenge是 HEX-encoded SHA256，不是 base64url"""
    return hashlib.sha256(code_verifier.encode("ascii")).hexdigest()

def get_auth_url(self, redirect_uri, state, code_verifier=None):
    params = {
        "client_key": self.credentials["client_key"],
        "redirect_uri": redirect_uri,
        "state": state,
        "scope": ",".join(self.required_scopes),
        "response_type": "code",
    }
    if code_verifier:
        params["code_challenge"] = _pkce_code_challenge(code_verifier)
        params["code_challenge_method"] = "S256"
    return f"{AUTH_URL}?{urlencode(params)}"

def exchange_code(self, code, redirect_uri, code_verifier=None):
    data = {
        "client_key": self.credentials["client_key"],
        "client_secret": self.credentials["client_secret"],
        "code": code,
        "grant_type": "authorization_code",
        "redirect_uri": redirect_uri,
    }
    if code_verifier:
        data["code_verifier"] = code_verifier
    resp = self._request("POST", TOKEN_URL, data=data)
    body = resp.json()
    if "access_token" not in body:
        raise OAuthError(f"TikTok token exchange failed: {body}", platform=self.platform_name)
    return OAuthTokens(access_token=body["access_token"], refresh_token=body.get("refresh_token"),
                      expires_in=body.get("expires_in"), scope=body.get("scope"), raw_response=body)
```

def refresh_token(self, short_lived_token: str) -> OAuthTokens:
    """Instagram/Facebook 长期Token交换"""
    resp = self._request("GET", f"{BASE_URL}/oauth/access_token", params={
        "grant_type": "fb_exchange_token",
        "client_id": self.credentials["client_id"],
        "client_secret": self.credentials["client_secret"],
        "fb_exchange_token": short_lived_token,
    })
    data = resp.json()
    if "access_token" not in data:
        raise OAuthError("Instagram long-lived token exchange failed", platform=self.platform_name)
    return OAuthTokens(access_token=data["access_token"], expires_in=data.get("expires_in"))

## 三、错误处理和重试机制

### 3.1 统一错误处理框架

```python
def _request(self, method, url, *, access_token=None, headers=None, params=None, json=None, data=None, files=None, timeout=REQUEST_TIMEOUT):
    req_headers = {}
    if access_token: req_headers["Authorization"] = f"Bearer {access_token}"
    if headers: req_headers.update(headers)
    with httpx.Client(timeout=timeout) as client:
        response = client.request(method, url, headers=req_headers, params=params, json=json, files=files)
    if response.status_code == 429:
        retry_after = response.headers.get("Retry-After")
        raise RateLimitError(f"Rate limit exceeded for {self.platform_name}: {response.text[:500]}",
            retry_after=int(retry_after) if retry_after else None, platform=self.platform_name)
    if response.status_code >= 400:
        raise APIError(f"{self.platform_name} API error {response.status_code}: {response.text[:500]}",
            status_code=response.status_code, platform=self.platform_name)
    return response
```
