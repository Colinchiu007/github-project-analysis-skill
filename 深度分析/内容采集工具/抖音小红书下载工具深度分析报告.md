```python
async def fetch_data_stream(url, request, headers, file_path):
    async with httpx.AsyncClient() as client:
        async with client.stream("GET", url, headers=headers) as response:
            async with aiofiles.open(file_path, 'wb') as out_file:
                async for chunk in response.aiter_bytes():
                    if await request.is_disconnected():
                        await out_file.close()
                        os.remove(file_path)
                        return False
                    await out_file.write(chunk)
            return True
```

### 5.1 异步 HTTP 客户端（TypeScript）

```typescript
class BaseCrawler {
  private client: AxiosInstance;
  private maxRetries: number;

  constructor(config: CrawlerConfig = {}) {
    this.maxRetries = config.maxRetries || 3;
    this.client = axios.create({
      timeout: (config.timeout || 10) * 1000,
      headers: config.headers || {},
    });
  }

  async get<T>(url: string): Promise<T> {
    for (let attempt = 0; attempt < this.maxRetries; attempt++) {
      try {
        const response = await this.client.get<T>(url);
        return response.data;
      } catch (error) {
        if (attempt === this.maxRetries - 1) throw error;
        await this.sleep(1000 * (attempt + 1));
      }
    }
    throw new Error('Max retries exceeded');
  }
}
```

### 5.5 文件格式检测

```typescript
const FILE_SIGNATURES: [number, Buffer, string][] = [
  [0, Buffer.from([0x89, 0x50, 0x4E, 0x47]), 'png'],
  [0, Buffer.from([0xFF, 0xD8, 0xFF]), 'jpeg'],
  [0, Buffer.from('RIFF'), 'webp'],
  [4, Buffer.from('ftypmp4'), 'mp4'],
];

function detectFileType(filePath: string): string {
  const fd = openSync(filePath, 'r');
  const buffer = Buffer.alloc(12);
  readSync(fd, buffer, 0, 12, 0);
  closeSync(fd);
  for (const [offset, signature, ext] of FILE_SIGNATURES) {
    if (buffer.subarray(offset, offset + signature.length).equals(signature)) return ext;
  }
  return 'bin';
}
```
