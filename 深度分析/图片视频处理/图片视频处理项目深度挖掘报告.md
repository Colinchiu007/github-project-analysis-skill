### 2.1 视频转 GIF（两遍调色板法）

```javascript
import { exec } from 'child_process';
import { promisify } from 'util';
import fs from 'fs/promises';
import path from 'path';
import os from 'os';

const execAsync = promisify(exec);

class VideoToGifConverter {
  static async convert(inputPath, outputPath, options = {}) {
    const { fps = 10, width = 480, startTime = 0, endTime = null, loop = true } = options;
    const tmpDir = os.tmpdir();
    const palettePath = path.join(tmpDir, `palette_${Date.now()}.png`);

    try {
      const scaleFilter = `scale=${width}:-1:flags=lanczos`;
      const timeArgs = [];
      if (startTime > 0) timeArgs.push('-ss', startTime.toString());
      if (endTime) timeArgs.push('-to', endTime.toString());

      // 第一遍：生成优化调色板
      await execAsync(
        `ffmpeg -i "${inputPath}" ${timeArgs.join(' ')} ` +
        `-vf "fps=${fps},${scaleFilter},palettegen" "${palettePath}"`
      );

      // 第二遍：使用调色板生成 GIF
      await execAsync(
        `ffmpeg -i "${inputPath}" -i "${palettePath}" ` +
        `-filter_complex "fps=${fps},${scaleFilter}[x]; [x][1:v]paletteuse" ` +
        `-loop ${loop ? '0' : '-1'} ${timeArgs.join(' ')} "${outputPath}"`
      );

      return outputPath;
    } finally {
      try { await fs.unlink(palettePath); } catch {}
    }
  }
}
```

### 3.1 URL 签名工具

```javascript
import crypto from 'crypto';

class ImgProxySigner {
  constructor(key, salt) {
    this.key = Buffer.from(key, 'hex');
    this.salt = Buffer.from(salt, 'hex');
  }

  sign(url, path = '/insecure/', expire = null) {
    const targetUrl = expire ? `${url}?expire=${expire}` : url;
    const hmac = crypto.createHmac('sha256', this.key);
    hmac.update(this.salt);
    hmac.update(targetUrl);
    const signature = hmac.digest('base64url');
    return `${path}${signature}/${targetUrl}`;
  }

  generateSignedUrl(imgproxyBaseUrl, imageUrl, options = {}) {
    const { width = null, height = null, resize = 'fit', format = 'auto', quality = null } = options;
    const processingOptions = [];
    if (width) processingOptions.push(`w:${width}`);
    if (height) processingOptions.push(`h:${height}`);
    if (resize !== 'fit') processingOptions.push(`rt:${resize}`);
    if (format !== 'auto') processingOptions.push(`f:${format}`);
    if (quality) processingOptions.push(`q:${quality}`);
    const path = processingOptions.length > 0
      ? `/insecure/rs:${resize}:${processingOptions.join('/')}/`
      : '/insecure/';
    return this.sign(imageUrl, path);
  }

  verify(url, signature) {
    const hmac = crypto.createHmac('sha256', this.key);
    hmac.update(this.salt);
    hmac.update(url);
    return signature === hmac.digest('base64url');
  }
}
```

### 3.2 源 URL 验证工具

```javascript
class SourceUrlValidator {
  constructor(options = {}) {
    this.allowedDomains = options.allowedDomains || [];
    this.blockedPatterns = options.blockedPatterns || [];
    this.maxUrlLength = options.maxUrlLength || 2048;
  }

  validate(url) {
    if (url.length > this.maxUrlLength) return { valid: false, error: 'URL too long' };
    let parsed;
    try { parsed = new URL(url); } catch { return { valid: false, error: 'Invalid URL format' }; }
    if (!['http:', 'https:'].includes(parsed.protocol)) return { valid: false, error: 'Only HTTP/HTTPS allowed' };
    if (this.allowedDomains.length > 0) {
      const isAllowed = this.allowedDomains.some(domain => 
        parsed.hostname === domain || parsed.hostname.endsWith(`.${domain}`)
      );
      if (!isAllowed) return { valid: false, error: 'Domain not allowed' };
    }
    for (const pattern of this.blockedPatterns) {
      if (pattern.test(url)) return { valid: false, error: 'URL matches blocked pattern' };
    }
    if (this.isPrivateIP(parsed.hostname)) return { valid: false, error: 'Private IP not allowed' };
    return { valid: true };
  }

  isPrivateIP(hostname) {
    const privatePatterns = [/^127\./, /^10\./, /^172\.(1[6-9]|2\d|3[01])\./, /^192\.168\./, /^0\./, /^localhost$/i];
    return privatePatterns.some(pattern => pattern.test(hostname));
  }
}
```

### 3.3 请求限流中间件

```javascript
class RateLimiter {
  constructor(options = {}) {
    this.windowMs = options.windowMs || 60000;
    this.maxRequests = options.maxRequests || 60;
    this.requests = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), this.windowMs);
  }

  checkLimit(identifier) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    if (!this.requests.has(identifier)) this.requests.set(identifier, []);
    const timestamps = this.requests.get(identifier).filter(ts => ts > windowStart);
    this.requests.set(identifier, timestamps);
    if (timestamps.length >= this.maxRequests) {
      return { allowed: false, remaining: 0, retryAfter: Math.ceil((timestamps[0] + this.windowMs - now) / 1000) };
    }
    timestamps.push(now);
    return { allowed: true, remaining: this.maxRequests - timestamps.length };
  }

  middleware() {
    return (req, res, next) => {
      const identifier = req.ip;
      const result = this.checkLimit(identifier);
      res.set('X-RateLimit-Limit', this.maxRequests.toString());
      res.set('X-RateLimit-Remaining', result.remaining.toString());
      if (!result.allowed) return res.status(429).json({ error: 'Too many requests' });
      next();
    };
  }

  cleanup() {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    for (const [identifier, timestamps] of this.requests) {
      const valid = timestamps.filter(ts => ts > windowStart);
      if (valid.length === 0) this.requests.delete(identifier);
      else this.requests.set(identifier, valid);
    }
  }
}
```

```javascript
import sharp from 'sharp';

class SmartImageCompressor {
  static async compressForPurpose(inputPath, outputPath, purpose = 'web') {
    const metadata = await sharp(inputPath).metadata();
    const presets = {
      web: { jpeg: { quality: 80, mozjpeg: true, progressive: true }, png: { compressionLevel: 6 } },
      thumbnail: { jpeg: { quality: 70, mozjpeg: true } },
      print: { jpeg: { quality: 95, chromaSubsampling: '4:4:4' } },
      archive: { jpeg: { quality: 60, mozjpeg: true } },
    };
    const preset = presets[purpose] || presets.web;
    const format = metadata.format;
    const pipeline = sharp(inputPath);
    if (format === 'jpeg' || format === 'jpg') pipeline.jpeg(preset.jpeg);
    else if (format === 'png') pipeline.png(preset.png);
    await pipeline.toFile(outputPath);
  }

  static async compressToTargetSize(inputPath, outputPath, targetSizeKB, format = 'jpeg') {
    const targetBytes = targetSizeKB * 1024;
    let low = 1, high = 100, bestQuality = 50;
    while (low <= high) {
      const mid = Math.floor((low + high) / 2);
      const tempPath = outputPath + '.temp';
      await sharp(inputPath).jpeg({ quality: mid, mozjpeg: true }).toFile(tempPath);
      const { size } = await fs.stat(tempPath);
      if (Math.abs(size - targetBytes) < targetBytes * 0.05) {
        await fs.rename(tempPath, outputPath);
        return { quality: mid, size };
      }
      if (size > targetBytes) high = mid - 1;
      else { low = mid + 1; bestQuality = mid; }
      await fs.unlink(tempPath).catch(() => {});
    }
    await sharp(inputPath).jpeg({ quality: bestQuality, mozjpeg: true }).toFile(outputPath);
    return { quality: bestQuality };
  }
}
```

```javascript
    }).png().toBuffer();

      .composite([{ input: semiTransparent, gravity: 'southeast', blend: 'over' }])
      .toFile(outputPath);
  }
```

```javascript
      const time = interval * i;
      const outputPath = path.join(outputDir, `thumb_${i}.jpg`);
      await this.extractThumbnail(videoPath, outputPath, {

        width,
```

## 七、图片缓存策略

### 7.1 双层缓存系统

```javascript
import fs from 'fs/promises';
import path from 'path';
import crypto from 'crypto';

class ImageCacheSystem {
  constructor(options = {}) {
    this.memoryCache = new Map();
    this.memoryMaxSize = (options.memoryMaxSize || 100) * 1024 * 1024;
    this.memoryCurrentSize = 0;
    this.diskCacheDir = options.diskCacheDir || './cache/images';
    this.diskMaxSize = options.diskMaxSize || 500 * 1024 * 1024;
    this.ttl = options.ttl || 3600000;
  }

  generateKey(imagePath, options) {
    const content = JSON.stringify({ path: imagePath, options });
    return crypto.createHash('md5').update(content).digest('hex');
  }

  async get(key) {
    // 内存缓存
    const memEntry = this.memoryCache.get(key);
    if (memEntry && Date.now() - memEntry.timestamp < this.ttl) {
      return memEntry.data;
    }
    // 磁盘缓存
    try {
      const filePath = path.join(this.diskCacheDir, `${key}.cache`);
      const data = await fs.readFile(filePath);
      const stat = await fs.stat(filePath);
      if (Date.now() - stat.mtimeMs < this.ttl) {
        this.memoryCache.set(key, { data, size: data.length, timestamp: Date.now() });
        this.memoryCurrentSize += data.length;
        return data;
      }
    } catch {}
    return null;
  }

  async set(key, data) {
    const size = Buffer.isBuffer(data) ? data.length : Buffer.byteLength(data);
    // 写入内存缓存
    this.memoryCache.set(key, { data, size, timestamp: Date.now() });
    this.memoryCurrentSize += size;
    this.evictMemory();
    // 写入磁盘缓存
    try {
      await fs.mkdir(this.diskCacheDir, { recursive: true });
      await fs.writeFile(path.join(this.diskCacheDir, `${key}.cache`), data);
    } catch {}
  }

  evictMemory() {
    while (this.memoryCurrentSize > this.memoryMaxSize) {
      const oldestKey = this.memoryCache.keys().next().value;
      if (oldestKey) {
        const entry = this.memoryCache.get(oldestKey);
        this.memoryCurrentSize -= entry.size;
        this.memoryCache.delete(oldestKey);
      } else break;
    }
  }

  async clear() {
    this.memoryCache.clear();
    try {
      const files = await fs.readdir(this.diskCacheDir);
      for (const file of files) await fs.unlink(path.join(this.diskCacheDir, file)).catch(() => {});
    } catch {}
  }
}
```

```javascript
import sharp from 'sharp';

class ColorAnalyzer {
  static async getDominantColors(imagePath, count = 5) {
    const { data, info } = await sharp(imagePath).resize(100, 100, { fit: 'inside' }).raw().toBuffer({ resolveWithObject: true });
    const colors = new Map();
    for (let i = 0; i < data.length; i += info.channels) {
      const qr = Math.round(data[i] / 51) * 51;
      const qg = Math.round(data[i + 1] / 51) * 51;
      const qb = Math.round(data[i + 2] / 51) * 51;
      const key = `${qr},${qg},${qb}`;
      colors.set(key, (colors.get(key) || 0) + 1);
    }
    const totalPixels = info.width * info.height;
    return [...colors.entries()]
      .sort((a, b) => b[1] - a[1])
      .slice(0, count)
      .map(([color, cnt]) => {
        const [r, g, b] = color.split(',').map(Number);
        return {
          rgb: { r, g, b },
          hex: `#${r.toString(16).padStart(2, '0')}${g.toString(16).padStart(2, '0')}${b.toString(16).padStart(2, '0')}`,
          percentage: ((cnt / totalPixels) * 100).toFixed(1),
        };
      });
  }

  static async isDarkImage(imagePath, threshold = 128) {
    const stats = await sharp(imagePath).greyscale().resize(50, 50, { fit: 'inside' }).stats();
    return { isDark: stats.channels[0].mean < threshold, brightness: stats.channels[0].mean };
  }
}
```

### 8.3 图片对比工具

```javascript
import sharp from 'sharp';

class ImageComparator {
  static async compare(imagePath1, imagePath2, options = {}) {
    const { threshold = 0.1 } = options;
    const [meta1, meta2] = await Promise.all([sharp(imagePath1).metadata(), sharp(imagePath2).metadata()]);
    if (meta1.width !== meta2.width || meta1.height !== meta2.height) {
      return { match: false, error: 'Different dimensions' };
    }
    const [img1, img2] = await Promise.all([sharp(imagePath1).raw().toBuffer(), sharp(imagePath2).raw().toBuffer()]);
    const channels = meta1.channels || 3;
    const totalPixels = meta1.width * meta1.height;
    let diffPixels = 0;
    for (let i = 0; i < img1.length; i += channels) {
      let pixelDiff = 0;
      for (let c = 0; c < channels; c++) pixelDiff += Math.abs(img1[i + c] - img2[i + c]);
      if (pixelDiff / channels > threshold * 255) diffPixels++;
    }
    return {
      match: diffPixels === 0,
      similarity: ((totalPixels - diffPixels) / totalPixels * 100).toFixed(2),
      diffPixels,
      totalPixels,
    };
  }

  static async calculatePSNR(imagePath1, imagePath2) {
    const [meta1, meta2] = await Promise.all([sharp(imagePath1).metadata(), sharp(imagePath2).metadata()]);
    if (meta1.width !== meta2.width || meta1.height !== meta2.height) throw new Error('Different dimensions');
    const [img1, img2] = await Promise.all([sharp(imagePath1).raw().toBuffer(), sharp(imagePath2).raw().toBuffer()]);
    let mse = 0;
    for (let i = 0; i < img1.length; i++) { const diff = img1[i] - img2[i]; mse += diff * diff; }
    mse /= (meta1.width * meta1.height * (meta1.channels || 3));
    if (mse === 0) return { psnr: Infinity, quality: 'Perfect' };
    const psnr = 10 * Math.log10((255 * 255) / mse);
    return { psnr: psnr.toFixed(2), mse: mse.toFixed(4), quality: psnr > 40 ? 'Excellent' : psnr > 30 ? 'Good' : 'Poor' };
  }
}
```
