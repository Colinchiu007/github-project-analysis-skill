## 二、sharp 核心 API 深度分析

### 2.1 架构设计：流水线模式 (Pipeline Pattern)

sharp 的核心设计理念是 **链式调用 + 延迟执行**。所有操作先注册到 `this.options`，最后通过 `_pipeline()` 一次性提交给 libvips C++ 层执行。

```typescript
sharp(input)
  .resize(200, 200, { fit: 'cover', position: 'centre' })
  .rotate(90)
  .composite([{ input: watermark, gravity: 'southeast' }])
  .jpeg({ quality: 85 })
  .toBuffer();
```

### 4.1 图片处理核心工具函数

```typescript
import sharp from 'sharp';

export interface ProcessImageOptions {
  input: Buffer | string;
  resize?: {
    width?: number;
    height?: number;
    fit?: 'cover' | 'contain' | 'fill' | 'inside' | 'outside';
    position?: string;
    withoutEnlargement?: boolean;
  };
  watermark?: {
    type: 'text' | 'image';
    text?: string;
    imagePath?: string;
    position: 'north' | 'south' | 'east' | 'west' | 'center'
           | 'northeast' | 'northwest' | 'southeast' | 'southwest';
    opacity?: number;
    fontSize?: number;
    fontColor?: string;
  };
  format: 'jpeg' | 'png' | 'webp' | 'gif';
  quality?: number;
}

export async function processImage(options: ProcessImageOptions): Promise<Buffer> {
  let pipeline = sharp(options.input);
  pipeline = pipeline.rotate();
  
  if (options.resize) {
    pipeline = pipeline.resize({
      width: options.resize.width,
      height: options.resize.height,
      fit: options.resize.fit || 'cover',
      position: options.resize.position || 'centre',
      withoutEnlargement: options.resize.withoutEnlargement ?? true,
      kernel: 'lanczos3'
    });
  }
  
  if (options.watermark) {
    const overlay = await createWatermarkOverlay(options.watermark);
    pipeline = pipeline.composite([{
      input: overlay,
      gravity: options.watermark.position,
      blend: 'over'
    }]);
  }
  
  switch (options.format) {
    case 'jpeg':
      pipeline = pipeline.jpeg({ quality: options.quality ?? 85, mozjpeg: true });
      break;
    case 'png':
      pipeline = pipeline.png({ quality: options.quality ?? 90, compressionLevel: 6 });
      break;
    case 'webp':
      pipeline = pipeline.webp({ quality: options.quality ?? 80, effort: 4 });
      break;
    case 'gif':
      pipeline = pipeline.gif({ effort: 7, dither: 1.0 });
      break;
  }
  
  return pipeline.toBuffer();
}
```

### 4.2 批量处理队列

```typescript
import { EventEmitter } from 'events';
import { processImage, ProcessImageOptions } from './image-processor';

export interface BatchTask {
  id: string;
  input: Buffer | string;
  options: ProcessImageOptions;
  status: 'pending' | 'processing' | 'completed' | 'failed';
  result?: Buffer;
  error?: Error;
}

export class BatchProcessor extends EventEmitter {
  private queue: BatchTask[] = [];
  private processing = 0;
  private concurrency: number;
  
  constructor(options: { concurrency?: number } = {}) {
    super();
    this.concurrency = options.concurrency ?? 4;
  }
  
  addTask(input: Buffer | string, options: ProcessImageOptions): string {
    const id = `task_${Date.now()}_${Math.random().toString(36).slice(2)}`;
    const task: BatchTask = { id, input, options, status: 'pending' };
    this.queue.push(task);
    this.processNext();
    return id;
  }
  
  private async processNext(): Promise<void> {
    if (this.processing >= this.concurrency) return;
    const task = this.queue.find(t => t.status === 'pending');
    if (!task) return;
    task.status = 'processing';
    this.processing++;
    try {
      task.result = await processImage(task.options);
      task.status = 'completed';
      this.emit('taskComplete', task);
    } catch (error) {
      task.status = 'failed';
      task.error = error as Error;
      this.emit('taskFailed', task);
    } finally {
      this.processing--;
      this.emit('progress', this.getProgress());
      this.processNext();
    }
  }
  
  getProgress() {
    return {
      total: this.queue.length,
      completed: this.queue.filter(t => t.status === 'completed').length,
      failed: this.queue.filter(t => t.status === 'failed').length,
      pending: this.queue.filter(t => t.status === 'pending').length,
    };
  }
  
  cancelAll(): void {
    for (const task of this.queue) {
      if (task.status === 'pending') {
        task.status = 'failed';
        task.error = new Error('Cancelled');
      }
    }
  }
}
```

### 4.3 多平台图片适配器

```typescript
import sharp from 'sharp';
import { processImage } from '../utils/image-processor';
import { PLATFORM_SPECS } from '../config/platform-sizes';

export interface PublishImage {
  platform: string;
  buffer: Buffer;
  filename: string;
  contentType: string;
}

export class PlatformAdapter {
  static async adapt(
    sourceImage: Buffer,
    platform: string,
    contentType: 'cover' | 'content'
  ): Promise<PublishImage> {
    const spec = PLATFORM_SPECS[platform];
    if (!spec) throw new Error(`Unknown platform: ${platform}`);
    const target = contentType === 'cover' ? spec.cover : spec.content;
    
    const metadata = await sharp(sourceImage).metadata();
    const sourceRatio = metadata.width! / metadata.height!;
    const targetRatio = target.width / (target.height || target.width);
    
    const resizeOptions = Math.abs(sourceRatio - targetRatio) < 0.2
      ? { width: target.width, height: target.height || target.width, fit: 'cover' as const, position: 'attention' }
      : { width: target.width, height: target.height || target.width, fit: 'contain' as const };
    
    const processed = await processImage({
      input: sourceImage,
      resize: resizeOptions,
      format: 'jpeg',
      quality: 85
    });
    
    return {
      platform,
      buffer: processed,
      filename: `${platform}_${contentType}_${Date.now()}.jpeg`,
      contentType: 'image/jpeg'
    };
  }
  
  static async adaptForMultiple(
    sourceImage: Buffer,
    platforms: string[],
    contentType: 'cover' | 'content' = 'cover'
  ): Promise<PublishImage[]> {
    return Promise.all(platforms.map(p => this.adapt(sourceImage, p, contentType)));
  }
}
```

### 4.4 格式转换模块

```typescript
import sharp from 'sharp';

export class FormatConverter {
  static async toJpeg(input: Buffer, quality = 85): Promise<Buffer> {
    return sharp(input).jpeg({ quality, mozjpeg: true }).toBuffer();
  }
  
  static async toPng(input: Buffer, quality = 90): Promise<Buffer> {
    return sharp(input).png({ quality, compressionLevel: 6 }).toBuffer();
  }
  
  static async toWebp(input: Buffer, quality = 80): Promise<Buffer> {
    return sharp(input).webp({ quality, effort: 4 }).toBuffer();
  }
  
  static async smartConvert(
    input: Buffer,
    maxFileSize?: number
  ): Promise<{ buffer: Buffer; format: string }> {
    const metadata = await sharp(input).metadata();
    const hasAlpha = metadata.channels === 4;
    const isAnimated = (metadata.pages ?? 0) > 1;
    
    if (isAnimated) {
      const webp = await this.toWebp(input, 80);
      if (!maxFileSize || webp.byteLength <= maxFileSize) {
        return { buffer: webp, format: 'webp' };
      }
    }
    
    if (hasAlpha) {
      const png = await this.toPng(input);
      return { buffer: png, format: 'png' };
    }
    
    const jpeg = await this.toJpeg(input, 85);
    return { buffer: jpeg, format: 'jpeg' };
  }
}
```

## 五、Electron 截图功能实现

```typescript
import { desktopCapturer, BrowserWindow, screen } from 'electron';
import sharp from 'sharp';

export class ScreenshotManager {
  async captureFullScreen(): Promise<Buffer> {
    const primaryDisplay = screen.getPrimaryDisplay();
    const { width, height } = primaryDisplay.size;
    const sources = await desktopCapturer.getSources({
      types: ['screen'],
      thumbnailSize: { width, height }
    });
    const primarySource = sources.find(s => 
      s.display_id === String(primaryDisplay.id)
    ) || sources[0];
    return primarySource.thumbnail.toPNG();
  }
  
  async captureRegion(region: { x: number; y: number; width: number; height: number }): Promise<Buffer> {
    const fullScreenshot = await this.captureFullScreen();
    return sharp(fullScreenshot)
      .extract({ left: region.x, top: region.y, width: region.width, height: region.height })
      .png()
      .toBuffer();
  }
}
```

1. **sharp 是核心引擎**：链式调用、流水线模式、丰富的 resize/composite/format API
