### 2.1 授权码流程

```typescript
// OAuth Provider 接口定义
interface OAuthProvider {
  id: string;
  name: string;
  type: 'oauth' | 'oidc';
  clientId: string;
  clientSecret: string;
  authorizationUrl: string;
  tokenUrl: string;
  userInfoUrl: string;
  scopes: string[];
  customParams?: Record<string, string>;
}

// 授权码流程完整实现
async function authorizationCodeFlow(provider: OAuthProvider): Promise<OAuthTokens> {
  const state = crypto.randomBytes(32).toString('hex');
  const codeVerifier = generateCodeVerifier();
  const codeChallenge = await generateCodeChallenge(codeVerifier);

  const authUrl = new URL(provider.authorizationUrl);
  authUrl.searchParams.set('client_id', provider.clientId);
  authUrl.searchParams.set('redirect_uri', getRedirectUri(provider));
  authUrl.searchParams.set('response_type', 'code');
  authUrl.searchParams.set('scope', provider.scopes.join(' '));
  authUrl.searchParams.set('state', state);
  authUrl.searchParams.set('code_challenge', codeChallenge);
  authUrl.searchParams.set('code_challenge_method', 'S256');

  const authCode = await openAuthWindow(authUrl.toString());

  const tokenResponse = await fetch(provider.tokenUrl, {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'authorization_code',
      code: authCode,
      redirect_uri: getRedirectUri(provider),
      client_id: provider.clientId,
      client_secret: provider.clientSecret,
      code_verifier: codeVerifier,
    }),
  });

  return tokenResponse.json();
}
```

### 2.2 PKCE 流程

```typescript
import * as crypto from 'crypto';

class PKCEHelper {
  static generateCodeVerifier(): string {
    return crypto.randomBytes(32)
      .toString('base64url')
      .replace(/[^a-zA-Z0-9]/g, '')
      .substring(0, 128);
  }

  static async generateCodeChallenge(verifier: string): Promise<string> {
    return crypto.createHash('sha256')
      .update(verifier)
      .digest('base64url');
  }

  static buildAuthorizationUrl(config: {
    authorizeEndpoint: string;
    clientId: string;
    redirectUri: string;
    scope: string;
    state: string;
    codeChallenge: string;
  }): string {
    const url = new URL(config.authorizeEndpoint);
    url.searchParams.set('response_type', 'code');
    url.searchParams.set('client_id', config.clientId);
    url.searchParams.set('redirect_uri', config.redirectUri);
    url.searchParams.set('scope', config.scope);
    url.searchParams.set('state', config.state);
    url.searchParams.set('code_challenge', config.codeChallenge);
    url.searchParams.set('code_challenge_method', 'S256');
    return url.toString();
  }
}
```

## 三、Token 安全存储

### 3.1 Electron 安全存储

```typescript
import { safeStorage } from 'electron';
import * as fs from 'fs/promises';
import * as path from 'path';

class SecureTokenStorage {
  private storagePath: string;
  private cache: Map<string, OAuthTokens> = new Map();

  constructor(appName: string = 'Multi-Publish') {
    const userDataPath = app.getPath('userData');
    this.storagePath = path.join(userDataPath, 'tokens');
  }

  async storeTokens(accountId: string, tokens: OAuthTokens): Promise<void> {
    this.cache.set(accountId, tokens);
    if (safeStorage.isEncryptionAvailable()) {
      const encrypted = safeStorage.encryptString(JSON.stringify(tokens));
      await fs.writeFile(this.getFilePath(accountId), encrypted);
    } else {
      const encrypted = Buffer.from(JSON.stringify(tokens)).toString('base64');
      await fs.writeFile(this.getFilePath(accountId), encrypted);
    }
  }

  async getTokens(accountId: string): Promise<OAuthTokens | null> {
    if (this.cache.has(accountId)) return this.cache.get(accountId)!;
    try {
      const encrypted = await fs.readFile(this.getFilePath(accountId));
      let decrypted: string;
      if (safeStorage.isEncryptionAvailable()) {
        decrypted = safeStorage.decryptString(encrypted);
      } else {
        decrypted = Buffer.from(encrypted.toString(), 'base64').toString();
      }
      const tokens = JSON.parse(decrypted);
      this.cache.set(accountId, tokens);
      return tokens;
    } catch { return null; }
  }

  async deleteTokens(accountId: string): Promise<void> {
    this.cache.delete(accountId);
    try { await fs.unlink(this.getFilePath(accountId)); } catch {}
  }

  isTokenExpired(tokens: OAuthTokens, bufferSeconds: number = 300): boolean {
    if (!tokens.expires_at) return false;
    return Date.now() >= (tokens.expires_at - bufferSeconds * 1000);
  }

  private getFilePath(accountId: string): string {
    const safeName = accountId.replace(/[^a-zA-Z0-9]/g, '_');
    return path.join(this.storagePath, `${safeName}.enc`);
  }
}
```

### 3.2 Token 缓存管理器

```typescript
class TokenCache {
  private cache: Map<string, CachedToken> = new Map();
  private refreshTimers: Map<string, NodeJS.Timeout> = new Map();

  set(accountId: string, tokens: OAuthTokens, refreshCallback: () => Promise<OAuthTokens>): void {
    const expiresAt = tokens.expires_at || Date.now() + 3600000;
    const refreshAt = expiresAt - 5 * 60 * 1000;
    this.cache.set(accountId, { tokens, expiresAt, refreshPromise: null });
    if (this.refreshTimers.has(accountId)) clearTimeout(this.refreshTimers.get(accountId)!);
    const delay = Math.max(0, refreshAt - Date.now());
    const timer = setTimeout(async () => {
      try {
        const newTokens = await refreshCallback();
        this.set(accountId, newTokens, refreshCallback);
      } catch { this.cache.delete(accountId); }
    }, delay);
    this.refreshTimers.set(accountId, timer);
  }

  async get(accountId: string): Promise<OAuthTokens | null> {
    const cached = this.cache.get(accountId);
    if (!cached) return null;
    if (Date.now() >= cached.expiresAt - 5 * 60 * 1000) {
      if (cached.refreshPromise) {
        await cached.refreshPromise;
        return this.cache.get(accountId)?.tokens || null;
      }
    }
    return cached.tokens;
  }
}

interface CachedToken {
  tokens: OAuthTokens;
  expiresAt: number;
  refreshPromise: Promise<OAuthTokens> | null;
}
```

```typescript
  async addAccount(config: {
    platform: PlatformType;
    tokens: OAuthTokens;
    profile: PlatformProfile;
  }): Promise<PlatformAccount> {
    const accountId = `${config.platform}:${config.profile.id}`;

      platform: config.platform,
      accountId: config.profile.id,
      username: config.profile.username,

      status: 'active',
      createdAt: new Date(),
      updatedAt: new Date(),
    };
    this.accounts.set(accountId, account);
    await this.persistAccount(account);
    return account;
  }

  getAccountsByPlatform(platform: PlatformType): PlatformAccount[] {
    return Array.from(this.accounts.values()).filter(a => a.platform === platform);
  }

  async validateAccount(accountId: string): Promise<boolean> {
    const account = this.accounts.get(accountId);
    if (!account) return false;
    if (this.isTokenExpired(account.tokens)) {
      try {
        const newTokens = await this.refreshAccountTokens(account);
        account.tokens = newTokens;
        account.status = 'active';
        account.updatedAt = new Date();
        await this.persistAccount(account);
        return true;
      } catch {
        account.status = 'expired';
        await this.persistAccount(account);
        return false;
      }
    }
    return true;
  }
}
```

### 5.1 会话管理器

```typescript
class SessionManager {
  private sessions: Map<string, AppSession> = new Map();
  private maxSessions: number = 50;

  async createSession(userId: string): Promise<AppSession> {
    const session: AppSession = {
      id: crypto.randomBytes(32).toString('hex'),
      userId,
      token: crypto.randomBytes(64).toString('base64url'),
      expiresAt: new Date(Date.now() + 24 * 60 * 60 * 1000),
      createdAt: new Date(),
      updatedAt: new Date(),
    };
    const userSessions = this.getSessionsByUser(userId);
    if (userSessions.length >= this.maxSessions) {
      const oldest = userSessions.sort((a, b) => a.createdAt.getTime() - b.createdAt.getTime())[0];
      this.sessions.delete(oldest.id);
    }
    this.sessions.set(session.id, session);
    await this.persistSession(session);
    return session;
  }

  async validateSession(token: string): Promise<AppSession | null> {
    const session = Array.from(this.sessions.values()).find(s => s.token === token);
    if (!session) return null;
    if (session.expiresAt < new Date()) {
      this.sessions.delete(session.id);
      return null;
    }
    session.updatedAt = new Date();
    return session;
  }

  async destroySession(sessionId: string): Promise<void> {
    this.sessions.delete(sessionId);
    await this.deleteSession(sessionId);
  }

  async destroyAllUserSessions(userId: string, exceptSessionId?: string): Promise<void> {
    const userSessions = this.getSessionsByUser(userId);
    for (const session of userSessions) {
      if (session.id !== exceptSessionId) {
        this.sessions.delete(session.id);
        await this.deleteSession(session.id);
      }
    }
  }
}

interface AppSession {
  id: string;
  userId: string;
  token: string;
  expiresAt: Date;
  createdAt: Date;
  updatedAt: Date;
}
```

### 6.1 RBAC 权限模型

```typescript
interface PermissionPolicy {
  subject: string;
  object: string;
  action: string;
  effect: 'allow' | 'deny';
}

class PermissionManager {
  private policies: PermissionPolicy[] = [];

  definePolicies(): void {
    this.policies = [
      { subject: 'user:*', object: 'account:*', action: 'read', effect: 'allow' },
      { subject: 'user:*', object: 'account:*', action: 'write', effect: 'allow' },
      { subject: 'user:*', object: 'platform:weibo', action: 'publish', effect: 'allow' },
      { subject: 'role:admin', object: '*', action: '*', effect: 'allow' },
    ];
  }

  checkPermission(subject: string, object: string, action: string): boolean {
    const matching = this.policies.filter(p =>
      this.matchPattern(p.subject, subject) &&
      this.matchPattern(p.object, object) &&
      this.matchPattern(p.action, action)
    );
    if (matching.some(p => p.effect === 'deny')) return false;
    return matching.some(p => p.effect === 'allow');
  }

  private matchPattern(pattern: string, value: string): boolean {
    if (pattern === '*') return true;
    if (pattern.includes('*')) {
      return new RegExp('^' + pattern.replace(/\*/g, '.*') + '$').test(value);
    }
    return pattern === value;
  }
}
```

## 七、综合 OAuth 客户端

```typescript
class OAuthClient {
  private provider: PlatformConfig;
  private storage: SecureTokenStorage;

  constructor(provider: PlatformConfig) {
    this.provider = provider;
    this.storage = new SecureTokenStorage();
  }

  async authorize(): Promise<OAuthTokens> {
    const codeVerifier = PKCEHelper.generateCodeVerifier();
    const codeChallenge = await PKCEHelper.generateCodeChallenge(codeVerifier);
    const state = crypto.randomBytes(32).toString('hex');

    const authUrl = PKCEHelper.buildAuthorizationUrl({
      authorizeEndpoint: this.provider.authorizationUrl,
      clientId: this.provider.clientId,
      redirectUri: this.provider.redirectUri,
      scope: this.provider.scopes.join(' '),
      state,
      codeChallenge,
    });

    const authCode = await this.openAuthWindow(authUrl);
    const tokens = await this.exchangeCode(authCode, codeVerifier);
    await this.storage.storeTokens(this.provider.id, tokens);
    return tokens;
  }

  async getValidToken(): Promise<string | null> {
    const tokens = await this.storage.getTokens(this.provider.id);
    if (!tokens) return null;
    if (this.storage.isTokenExpired(tokens)) {
      try {
        const newTokens = await this.refreshTokens(tokens.refresh_token!);
        await this.storage.storeTokens(this.provider.id, newTokens);
        return newTokens.access_token;
      } catch { return null; }
    }
    return tokens.access_token;
  }

  private async refreshTokens(refreshToken: string): Promise<OAuthTokens> {
    const response = await fetch(this.provider.tokenUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        grant_type: 'refresh_token',
        refresh_token: refreshToken,
        client_id: this.provider.clientId,
        client_secret: this.provider.clientSecret,
      }),
    });
    if (!response.ok) throw new Error(`Token refresh failed: ${response.status}`);
    return response.json();
  }

  async secureFetch(url: string, options: RequestInit = {}): Promise<Response> {
    const token = await this.getValidToken();
    if (!token) throw new Error('No valid token available');
    return fetch(url, {
      ...options,
      headers: { ...options.headers, 'Authorization': `Bearer ${token}` },
    });
  }
}
```

## 九、安全注意事项

1. **PKCE 强制要求**: 桌面应用必须使用 PKCE，防止授权码拦截
2. **Token 加密存储**: 使用 Electron safeStorage 或 OS 级加密
3. **Refresh Token 保护**: 只在主进程存储，不发送到渲染进程
4. **CSRF 防护**: 使用 state 参数，参考 better-auth 的 origin check
5. **会话超时**: 设置合理的会话过期时间，参考 logto 的 TTL 配置

## 十、总结

**最佳组合方案**:
- **认证流程**: better-auth 的 Provider 插件模式
- **Token 管理**: googleapis 的事件驱动自动刷新
- **存储安全**: Electron safeStorage + better-auth 加密模式
- **权限控制**: casbin 的 RBAC 模型
- **会话管理**: better-auth 的 Session 模型

**避免的做法**:
- 不要直接使用 passport（过于老旧，不适合现代 Electron 应用）
- 不要使用 oauth2-proxy（反向代理模式不适合桌面应用）
- 不要使用 supertokens-core（Java 实现，与 TypeScript 技术栈不匹配）
