### 1.3 v-model 双向绑定

```vue
<script setup>
import { useEditor, EditorContent } from '@tiptap/vue-3'
import StarterKit from '@tiptap/starter-kit'
import { watch } from 'vue'

const props = defineProps({ modelValue: String })
const emit = defineEmits(['update:modelValue'])

const editor = useEditor({
  content: props.modelValue,
  extensions: [StarterKit],
  onUpdate: ({ editor }) => {
    emit('update:modelValue', editor.getHTML())
  },
})

watch(() => props.modelValue, (value) => {
  const isSame = editor.value.getHTML() === value
  if (!isSame) editor.value.commands.setContent(value, false)
})
</script>
```

### 1.5 内容导出转换器

```javascript
export class ContentExporter {
  static toMarkdown(editor) {
    const json = editor.getJSON()
    return this.jsonToMarkdown(json)
  }

  static toPlainText(editor) {
    return editor.getText()
  }

  static toWechatHTML(editor) {
    const html = editor.getHTML()
    return html
      .replace(/<iframe[^>]*>.*?<\/iframe>/gi, '')
      .replace(/<script[^>]*>.*?<\/script>/gi, '')
      .replace(/style="[^"]*"/gi, '')
  }

    return text.length > 2000 ? text.slice(0, 2000) + '...' : text
  }
}
```

## 二、sharp — 高性能图片处理

### 2.1 各平台尺寸适配系统

```javascript
import sharp from 'sharp'

const PLATFORM_SIZES = {
  weibo: {
    image: { width: 1200, height: null, fit: 'inside' },
    avatar: { width: 400, height: 400, fit: 'cover' },
    cover: { width: 920, height: 300, fit: 'cover' },
  },
  wechat: {
    image: { width: 1080, height: null, fit: 'inside' },
    avatar: { width: 640, height: 640, fit: 'cover' },
    cover: { width: 900, height: 500, fit: 'cover' },
  },
  twitter: {
    image: { width: 1600, height: 900, fit: 'cover' },
    avatar: { width: 400, height: 400, fit: 'cover' },
    cover: { width: 1500, height: 500, fit: 'cover' },
  },
  zhihu: {
    image: { width: 1080, height: null, fit: 'inside' },
    avatar: { width: 200, height: 200, fit: 'cover' },
    cover: { width: 690, height: null, fit: 'inside' },
  },
}

export async function processImage(inputPath, platform, type = 'image') {
  const config = PLATFORM_SIZES[platform]?.[type] || PLATFORM_SIZES.weibo.image
  const pipeline = sharp(inputPath)
    .autoOrient()
    .resize({ width: config.width, height: config.height, fit: config.fit, withoutEnlargement: true })

  switch (platform) {
    case 'wechat': return pipeline.jpeg({ quality: 85, mozjpeg: true }).toBuffer()
    case 'weibo': return pipeline.jpeg({ quality: 80 }).toBuffer()
    case 'twitter': return pipeline.png({ compressionLevel: 6 }).toBuffer()
    default: return pipeline.webp({ quality: 85 }).toBuffer()
  }
}
```

### 2.2 批量处理系统

```javascript
export async function batchProcessImages(inputDir, outputDir, platform) {
  const files = await readdir(inputDir)
  const imageFiles = files.filter(f => /\.(jpg|jpeg|png|webp|avif|gif|tiff)$/i.test(f))
  const CONCURRENCY = 4
  const results = []

  for (let i = 0; i < imageFiles.length; i += CONCURRENCY) {
    const batch = imageFiles.slice(i, i + CONCURRENCY)
    const batchResults = await Promise.allSettled(
      batch.map(async (file) => {
        const processed = await processImage(join(inputDir, file), platform)
        await sharp(processed).toFile(join(outputDir, `${platform}_${file}`))
        return { file, status: 'success' }
      })
    )
    results.push(...batchResults)
  }

  return {
    total: imageFiles.length,
    success: results.filter(r => r.status === 'fulfilled').length,
    failed: results.filter(r => r.status === 'rejected').length,
  }
}
```

```javascript
export async function addWatermark(inputPath, outputPath, options = {}) {
  const { text = 'Multi-Publish', position = 'southeast', opacity = 0.6, fontSize = 24, color = 'white' } = options

  const watermarkSvg = Buffer.from(`
    <svg width="300" height="60">
      <style>
        .watermark { font-size: ${fontSize}px; font-family: Arial, sans-serif; fill: ${color}; fill-opacity: ${opacity}; font-weight: bold; }
      </style>
      <text x="10" y="40" class="watermark">${text}</text>
    </svg>
  `)

  const metadata = await sharp(inputPath).metadata()
  const positionMap = {
    northwest: { left: 10, top: 10 },
    northeast: { left: metadata.width - 310, top: 10 },
    southwest: { left: 10, top: metadata.height - 70 },
    southeast: { left: metadata.width - 310, top: metadata.height - 70 },
    centre: { left: Math.round((metadata.width - 300) / 2), top: Math.round((metadata.height - 60) / 2) },
  }

  return sharp(inputPath)
    .composite([{ input: watermarkSvg, ...(positionMap[position] || positionMap.southeast) }])
    .toFile(outputPath)
}
```

## 三、Playwright — 浏览器自动化

### 3.1 OAuth 登录流程

```javascript
import { chromium } from 'playwright'

export class OAuthManager {
  async startOAuthLogin(platform) {
    const userDataDir = `./browser-data/${platform}`
    const context = await chromium.launchPersistentContext(userDataDir, {
      headless: false,
      viewport: { width: 1280, height: 800 },
      args: ['--disable-blink-features=AutomationControlled'],
    })
    const page = await context.newPage()

    const authUrls = {
      github: 'https://github.com/login/oauth/authorize',
      weibo: 'https://api.weibo.com/oauth2/authorize',
    }

    await page.goto(authUrls[platform])
    try {
      await page.waitForURL('**/callback**', { timeout: 300000 })
      const code = new URL(page.url()).searchParams.get('code')
      const cookies = await context.cookies()
      await this.saveCookies(platform, cookies)
      return { success: true, code }
    } catch (error) {
      return { success: false, error: error.message }
    }
  }
}
```

## 五、axios vs ky — HTTP 客户端选择

### 5.1 对比分析

| 特性 | axios (106k stars) | ky (12k stars) |
|------|-------------------|----------------|
| 体积 | ~13KB | ~3KB |
| API 风格 | Promise-based 传统 | 现代 Fetch API 封装 |
| 拦截器 | 内置请求/响应拦截器 | hooks |
| 重试 | 需要插件 | 内置重试 |
| TypeScript | 良好支持 | 原生 TypeScript |

### 5.2 推荐方案

```typescript
import axios from 'axios'
import ky from 'ky'

// 内部 API 客户端（axios，复杂拦截器）
export const internalApi = axios.create({
  baseURL: 'http://localhost:3000/api',
  timeout: 30000,
})

// 平台 API 客户端（ky，轻量级 + 内置重试）
export const platformApi = ky.create({
  timeout: 30000,
  retry: { limit: 2 },
```

## 六、综合架构建议

### 6.1 技术栈组合

```
Electron + Vue 3 + Vite
├── 编辑器层：tiptap (富文本编辑 + 多平台适配)
├── 图片处理层：sharp (水印 + 尺寸适配 + 格式转换)
├── 自动化层：Playwright (OAuth + Cookie + 反检测发布)
├── 认证层：better-auth (用户认证 + 多平台 Token)
└── HTTP 层：axios + ky (API 通信)
```

### 6.2 关键架构决策

1. **图片处理放在主进程**：sharp 是原生模块，必须在 Electron 主进程中运行
2. **Playwright 隔离运行**：每个平台使用独立的浏览器上下文
3. **tiptap 扩展化**：为每个平台编写专用扩展
4. **Token 加密存储**：使用 Electron 的 safeStorage API

### 6.3 性能优化要点

- **sharp 批量处理**：并发控制 4-8 个，避免内存溢出
- **Playwright 复用上下文**：使用 `launchPersistentContext`
- **tiptap 节流更新**：编辑器内容更新使用 debounce（300ms）
- **axios/ky 缓存**：对 GET 请求启用缓存
