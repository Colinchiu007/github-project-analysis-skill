**ProxyTierTracker — 基于错误历史的智能代理切换：**

```typescript
class ProxyTierTracker {
  private histogram: number[];
  private currentTier: number;

  private processStep(): void {
    this.histogram.forEach((x, i) => {
      if (this.currentTier === i) return;
      if (x > 0) this.histogram[i]--;
    });
    const left = this.currentTier > 0 ? this.histogram[this.currentTier - 1] : Infinity;
    const right = this.currentTier < this.histogram.length - 1 
                  ? this.histogram[this.currentTier + 1] : Infinity;
    if (this.histogram[this.currentTier] > Math.min(left, right)) {
      this.currentTier = left <= right ? this.currentTier - 1 : this.currentTier + 1;
    } else if (this.histogram[this.currentTier] === left) {
      this.currentTier--;
    }
  }

  addError(tier: number) {
    this.histogram[tier] += 10;
  }

  predictTier() {
    this.processStep();
    return this.currentTier;
  }
}
```

async function cookie({ data, id }) {
  const hasPermission = await BrowserAPIService.permissions.contains({
    permissions: ['cookies'],
  });
  if (!hasPermission) throw new Error('no-permission');
  let key = data.type;
  if (key === 'get' && data.getAll) key = 'getAll';
  if (data.useJson) {
    const obj = parseJSON(data.jsonCode, null);
    if (!obj) throw new Error('Invalid JSON format');
    result = await BrowserAPIService.cookies[key](obj);
  } else {
    const values = getValues(data, keys[key]);
    if (values.expirationDate) {
      values.expirationDate = Date.now() / 1000 + +values.expirationDate;
    }
    if (data.type === 'remove' && !data.name) {
      const cookies = await BrowserAPIService.cookies.getAll({ url: data.url });
      await Promise.allSettled(
        cookies.map(({ name }) => BrowserAPIService.cookies.remove({ name, url: data.url }))
      );
      result = cookies;
    } else {
      result = await BrowserAPIService.cookies[key](values);
    }
  }
  return { data: result, nextBlockId: this.getBlockConnections(id) };

function setProxy({ data, id }) {
  return new Promise((resolve, reject) => {
    if (data.clearProxy) {
      BrowserAPIService.proxy.settings.clear({});
    }
    const config = {
      mode: 'fixed_servers',
      rules: {
        singleProxy: { scheme: data.scheme },
        bypassList: isWhitespace(data.bypassList) ? [] : data.bypassList.split(','),
      },
    };
    let proxyHost = data.host;
    const schemeRegex = /^https?|socks4|socks5/i;
    if (schemeRegex.test(data.host)) {
      let [scheme, host] = data.host.split(/:\/\/(.*)/);
      if (host.includes(':')) {
        [host, proxyPort] = host.split(':');
      }
      proxyHost = host;
      config.rules.singleProxy.scheme = scheme;
    }
    config.rules.singleProxy.host = proxyHost;
    config.rules.singleProxy.port = +proxyPort;
    BrowserAPIService.proxy.settings.set({ value: config, scope: 'regular' })
      .then(() => { resolve({ data: data.host, nextBlockId }); });
  });
}

  return new Promise((resolve, reject) => {
    let retryCount = 1;
    const replacedValue = {};
    const testAllConditions = async () => {
      try {
        for (let index = 0; index < data.conditions.length; index += 1) {

def _bezier(p0, p1, p2, p3, t):
    u = 1 - t
    return Point(
        u**3 * p0.x + 3 * u**2 * t * p1.x + 3 * u * t**2 * p2.x + t**3 * p3.x,

def _random_control_points(start, end):
    dx, dy = end.x - start.x, end.y - start.y
    dist = math.hypot(dx, dy) or 1
    px, py = -dy / dist, dx / dist
    bias1 = rand(-0.3, 0.3) * dist
    bias2 = rand(-0.3, 0.3) * dist
    return (
        Point(start.x + dx * 0.25 + px * bias1, start.y + dy * 0.25 + py * bias1),
