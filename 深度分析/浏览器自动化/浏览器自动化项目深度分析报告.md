### 2.3 等待机制

**Automa的waitTabLoaded**：

```javascript
function waitTabLoaded({ tabId, listenError = false, ms = 10000 }) {
  return new Promise((resolve, reject) => {
    let timeout = null;
    const excludeErrors = ['net::ERR_BLOCKED_BY_CLIENT', 'net::ERR_ABORTED'];
    const onCompleted = (details) => {
      if (details.tabId !== tabId || details.frameId !== 0) return;
      clearTimeout(timeout);
      BrowserAPIService.webNavigation.onCompleted.removeListener(onCompleted);
      resolve();
    };
    const onErrorOccurred = (details) => {
      if (excludeErrors.includes(details.error)) return;
      clearTimeout(timeout);
      reject(new Error(details.error));
    };
    if (ms > 0) {
      timeout = setTimeout(() => reject(new Error('Timeout')), ms);
    }
  });
}
```

### 3.2 CDP隔离世界（核心反检测技术）

**CloakBrowser的CDP隔离世界**：

```python
class _SyncIsolatedWorld:
    def __init__(self, page: Any):
        self._page = page
        self._cdp = None
        self._context_id = None
    
    def _create_world(self) -> int:
        cdp = self._ensure_cdp()
        tree = cdp.send("Page.getFrameTree")
        frame_id = tree["frameTree"]["frame"]["id"]
        result = cdp.send("Page.createIsolatedWorld", {
            "frameId": frame_id,
            "worldName": "",
            "grantUniveralAccess": True,
        })
        self._context_id = result["executionContextId"]
        return self._context_id
    
    def evaluate(self, expression: str) -> Any:
        if self._context_id is None:
            self._create_world()
        for attempt in range(2):
            try:
                result = self._cdp.send("Runtime.evaluate", {
                    "expression": expression,
                    "contextId": self._context_id,
                    "returnByValue": True,
                })
                if "exceptionDetails" in result:
                    if attempt == 0:
                        self._create_world()
                        continue
                    return None
                return result.get("result", {}).get("value")
            except Exception:
                if attempt == 0:
                    self._context_id = None
                    self._create_world()
                    continue
                return None
```

```python
def _ease_in_out(t: float) -> float:
    if t < 0.5:
        return 4 * t * t * t
    return 1 - pow(-2 * t + 2, 3) / 2

def _bezier(p0, p1, p2, p3, t) -> Point:
    u = 1 - t
    return Point(
        u**3 * p0.x + 3 * u**2 * t * p1.x + 3 * u * t**2 * p2.x + t**3 * p3.x,
        u**3 * p0.y + 3 * u**2 * t * p1.y + 3 * u * t**2 * p2.y + t**3 * p3.y,

def human_move(raw, start_x, start_y, end_x, end_y, cfg):
```

### 7.1 人类化鼠标移动（CloakBrowser）

```typescript
export async function humanMouseMove(
  page: any,
  startX: number, startY: number,
  endX: number, endY: number,
  options: { wobbleMax?: number; overshootChance?: number } = {}
) {
  const dist = Math.hypot(endX - startX, endY - startY);
  if (dist < 1) return;
  const steps = Math.max(25, Math.min(80, Math.round(dist / 8)));
  const [cp1, cp2] = randomControlPoints({ x: startX, y: startY }, { x: endX, y: endY });
  for (let i = 0; i <= steps; i++) {
    const progress = i / steps;
    const easedT = easeInOut(progress);
    const pt = bezier({ x: startX, y: startY }, cp1, cp2, { x: endX, y: endY }, easedT);
    const wobbleAmp = Math.sin(Math.PI * progress) * (options.wobbleMax ?? 1.5);
    const wx = pt.x + (Math.random() - 0.5) * 2 * wobbleAmp;
    const wy = pt.y + (Math.random() - 0.5) * 2 * wobbleAmp;
    await page.mouse.move(Math.round(wx), Math.round(wy));
    await sleep(random.uniform(8, 18));
  }
}
```

### 7.2 CDP隔离世界（CloakBrowser）

```typescript
export class IsolatedWorld {
  private page: any;
  private cdp: any = null;
  private contextId: number | null = null;
  
  async evaluate(expression: string): Promise<any> {
    if (this.contextId === null) await this.createWorld();
    for (let attempt = 0; attempt < 2; attempt++) {
      try {
        const result = await this.cdp.send('Runtime.evaluate', {
          expression, contextId: this.contextId, returnByValue: true,
        });
        if ('exceptionDetails' in result) {
          if (attempt === 0) { await this.createWorld(); continue; }
          return null;
        }
        return result.result?.value;
      } catch {
        if (attempt === 0) { this.contextId = null; await this.createWorld(); continue; }
        return null;
      }
    }
  }
}
```

```typescript
class BrowserPool {
  async newPage(platform: string, publishTask: () => Promise<void>): Promise<void> {
    const controller = this.pickOrLaunchBrowser(platform);
    const page = await controller.newPage();

      await publishTask();
```
