```typescript
import { app, BrowserWindow, protocol } from "electron";
import path from "path";

let mainWindow: BrowserWindow | null = null;

function createMainWindow(): Promise<void> {
  return new Promise((resolve) => {
    const win = new BrowserWindow({
      width: 1000, height: 700, minWidth: 800, minHeight: 500,
      frame: false, show: false, autoHideMenuBar: true,
    });
    mainWindow = win;
    win.once("ready-to-show", () => { win.show(); resolve(); });
    // 开发模式加载 Vite dev server，生产模式加载打包文件
    const isDev = process.env.NODE_ENV === "dev" || !app.isPackaged;
    if (process.env.VITE_DEV) {
      void win.loadURL("http://localhost:5173");
    } else {
      const htmlPath = isDev
        ? path.join(process.cwd(), "data", "web", "index.html")
        : path.join(app.getPath("userData"), "data", "web", "index.html");
      void win.loadFile(htmlPath);
    }
  });
}

app.whenReady().then(async () => {
  // 启动内嵌 Express 服务器
  const mod = requireWithCustomPaths(servePath);
  const port = await mod.default(true);
  process.env.PORT = port;
  // 注册自定义协议处理器
  protocol.handle("multipublish", (request) => {
    const url = new URL(request.url);
    const pathname = url.hostname.toLowerCase();
    const handlers = {
      getappurl: () => ({ url: `http://localhost:${port}/api` }),
      windowminimize: () => { mainWindow?.minimize(); return { ok: true }; },
      windowmaximize: () => {
        if (mainWindow?.isMaximized()) mainWindow.unmaximize();
        else mainWindow?.maximize();
        return { ok: true };
      },
      windowclose: () => { app.exit(0); return { ok: true }; },
    };
    const handler = handlers[pathname];
    return new Response(JSON.stringify(handler ? handler() : { error: "未知接口" }), {
      headers: { "Content-Type": "application/json" },
    });
  });
  await createMainWindow();
});
```

**ytDownloader（推荐 — 最简洁的 IPC 模式）**：

```javascript
const { app, BrowserWindow, ipcMain, Tray, Menu, clipboard, shell, dialog, session } = require("electron");

function registerIpcHandlers() {
  ipcMain.on("reload", () => { appState.mainWindow?.reload(); });
  ipcMain.on("quit", () => { app.quit(); });
  ipcMain.on("progress", (_event, percentage) => {
    appState.mainWindow?.setProgressBar(percentage);
  });
  ipcMain.handle("show-file", async (_event, fullPath) => {
    try {
      await fs.stat(fullPath);
      shell.showItemInFolder(fullPath);
      return { success: true };
    } catch (error) {
      return { success: false, error: error.message };
    }
  });
  ipcMain.handle("select-location-main", async () => {
    const { canceled, filePaths } = await dialog.showOpenDialog(
      appState.mainWindow, { properties: ["openDirectory"] }
    );
    if (!canceled && filePaths.length > 0) {
      appState.mainWindow.webContents.send("downloadPath", filePaths);
    }
  });
  ipcMain.handle("get-system-proxy", async (_event, targetUrl = "https://youtube.com") => {
    const proxyInfo = await session.defaultSession.resolveProxy(targetUrl);
    const rule = String(proxyInfo).split(";").map(s => s.trim())
      .find(s => s && s.toUpperCase() !== "DIRECT");
    if (!rule) return null;
    const [type, hostPort] = rule.split(/\s+/, 2);
    const protocol = { PROXY: "http://", HTTP: "http://", HTTPS: "https://",
      SOCKS: "socks5://", SOCKS5: "socks5://", SOCKS4: "socks4://" }[type.toUpperCase()];
    return protocol ? `${protocol}${hostPort}` : null;
  });
}
```

## 三、窗口管理

### 3.1 窗口状态持久化（ytDownloader）

```javascript
const USER_DATA_PATH = app.getPath("userData");
const CONFIG_FILE_PATH = path.join(USER_DATA_PATH, "multipublish.json");

async function loadConfiguration() {
  try {
    const fileContent = await fs.readFile(CONFIG_FILE_PATH, "utf8");
    appState.config = JSON.parse(fileContent);
  } catch {
    appState.config = { bounds: { width: 1024, height: 768 }, isMaximized: false };
  }
}

function createWindow() {
  const bounds = appState.config.bounds || {};
  appState.mainWindow = new BrowserWindow({ ...bounds, minWidth: 800, minHeight: 600, show: false });
  const saveBounds = () => {
    if (appState.mainWindow && !appState.mainWindow.isMaximized()) {
      appState.config.bounds = appState.mainWindow.getBounds();
    }
  };
  appState.mainWindow.on("resize", saveBounds);
  appState.mainWindow.on("move", saveBounds);
  appState.mainWindow.on("maximize", () => { appState.config.isMaximized = true; });
  appState.mainWindow.on("unmaximize", () => { appState.config.isMaximized = false; });
  appState.mainWindow.once("ready-to-show", () => {
    if (appState.config.isMaximized) appState.mainWindow.maximize();
    appState.mainWindow.show();
  });
}
```

### 3.2 系统托盘（ytDownloader）

```javascript
function createTray() {
  if (appState.tray) return;
  let iconPath;
  if (process.platform === "win32") iconPath = path.join(__dirname, "resources/icon.ico");
  else if (process.platform === "darwin") iconPath = path.join(__dirname, "resources/icons/16x16.png");
  else iconPath = path.join(__dirname, "resources/icons/256x256.png");
  appState.tray = new Tray(iconPath);
  const contextMenu = Menu.buildFromTemplate([
    { label: "显示主窗口", click: () => { appState.mainWindow?.show(); if (app.dock) app.dock.show(); }},
    { label: "粘贴链接快速发布", click: async () => {
      const text = clipboard.readText();
      appState.mainWindow?.show();
      appState.mainWindow?.webContents.send("quick-publish", text);
    }},
    { type: "separator" },
    { label: "退出", click: () => { app.quit(); } },
  ]);
  appState.tray.setToolTip("Multi-Publish");
  appState.tray.setContextMenu(contextMenu);
  appState.tray.on("click", () => { appState.mainWindow?.show(); });
}
```

## 四、自动更新

```javascript
autoUpdater.autoDownload = false;
autoUpdater.on("update-available", async (info) => {
  const { response } = await dialog.showMessageBox(appState.mainWindow, {
    type: "info", buttons: ["更新", "取消"],
    title: "发现新版本", message: `版本 ${info.version} 可用`,
  });
  if (response === 0) autoUpdater.downloadUpdate();
});
autoUpdater.on("update-downloaded", async () => {
  const { response } = await dialog.showMessageBox(appState.mainWindow, {
    type: "info", buttons: ["立即重启", "稍后"],
    title: "更新已下载", message: "新版本已准备就绪",
  });
  if (response === 0) autoUpdater.quitAndInstall(true, true);
});
autoUpdater.on("download-progress", (info) => {
  appState.mainWindow.webContents.send("update-progress", info.percent);
});
setTimeout(() => autoUpdater.checkForUpdates(), 5000);
setInterval(() => autoUpdater.checkForUpdates(), 12 * 60 * 60 * 1000);
```
